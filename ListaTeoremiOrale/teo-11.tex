\begin{theorem}[Weirestrass][BDPG,10.10]
  Sia $\fn$, Supponiamo che:
  \begin{itemize}
    \item[(i)] A sia limitato e chiuso, (in $n=1$, $A = [a,b], \p A = \{a,b\}, \overcirc{A} = (a,b)$)
    \item[(ii)] f sia continua su A 
  \end{itemize}
  Allora esiste $\min_{A}f$ e $\max_{A}f$
\end{theorem}
